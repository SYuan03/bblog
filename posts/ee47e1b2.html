SpringBoot实现购物车结算

package com.showmebug.checkout.service.impl;

import com.showmebug.checkout.dto.ShoppingCartDTO;
import com.showmebug.checkout.entity.Account;
import com.showmebug.checkout.entity.Goods;
import com.showmebug.checkout.exception.IllegalBalanceException;
import com.showmebug.checkout.exception.IllegalStockException;
import com.showmebug.checkout.mapper.AccountMapper;
import com.showmebug.checkout.mapper.GoodsMapper;
import com.showmebug.checkout.service.GoodsService;
import java.util.List;
import javax.annotation.Resource;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.jdbc.datasource.DataSourceTransactionManager;
import org.springframework.stereotype.Service;
import org.springframework.transaction.TransactionDefinition;
import org.springframework.transaction.TransactionStatus;
import org.springframework.transaction.annotation.Transactional;

/**
 * @author ShowMeBug出题官
 * 业务层实现
 */
@Service
public class GoodsServiceImpl implements GoodsService {

    @Resource
    private GoodsMapper goodsMapper;

    @Resource
    private AccountMapper accountMapper;

    /*(2)请在这里书写你的代码*/
    @Autowired
    DataSourceTransactionManager dataSourceTransactionManager;

    @Autowired
    TransactionDefinition transactionDefinition;

    private final Object lock = new Object();	// 一开始以为是没处理并发，问的GPT搞的

    @Override
    public boolean checkOut(ShoppingCartDTO shoppingCart) {
        /*(1)请在这里书写你的代码*/
        // 开启事务
        TransactionStatus transactionStatus = dataSourceTransactionManager.getTransaction(
            transactionDefinition
        );
        Account account = accountMapper.selectById(shoppingCart.getAccountId());
        List<Integer> goodsList = shoppingCart.getGoodsIdList();

        double moneyCount = 0;
        // 先检查库存是否满足
        for (int i = 0; i < goodsList.size(); i++) {
            synchronized (lock) {
                Goods goods = goodsMapper.selectById(goodsList.get(i));
                int sales = goods.getSales();
                int stock = goods.getStock();
                if (stock >= 1) {	// 漏了个=呃呃
                    moneyCount += goods.getPrice();
                    goods.setSales(sales + 1);
                    goods.setStock(stock - 1);
                    goodsMapper.update(goods);
                } else {
                    dataSourceTransactionManager.rollback(transactionStatus);
                    throw new IllegalStockException(); // 库存不足
                }
            }
        }

        if (account.getBalance() < moneyCount) {
            dataSourceTransactionManager.rollback(transactionStatus);
            throw new IllegalBalanceException();
        }

        account.setBalance(account.getBalance() - moneyCount);
        accountMapper.update(account);

        dataSourceTransactionManager.commit(transactionStatus);
        return true;
    }
}
